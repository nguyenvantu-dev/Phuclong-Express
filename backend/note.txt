CREATE PROCEDURE [dbo].[CongNo_DanhSach]
	(
	@UserName nvarchar(100) 
	,@Status int
	)
AS
BEGIN
	SELECT * FROM [CONGNO] 
	where ([UserName] = @UserName or @UserName is null) and [Status] = @Status
END;

CREATE PROCEDURE [dbo].[CongNo_Insert]
	@NoiDung nvarchar(100)
	,@NgayGhiNo date
	,@DR float
	,@CR float
	,@UserName nvarchar(100)
	,@GhiChu nvarchar(500)
	,@Status int
	,@LoHangID int
	,@NguoiTao nvarchar(250)
	,@LoaiPhatSinh int
AS
	
BEGIN
	DECLARE	@DuocCapNhat int
	EXEC @DuocCapNhat = [SP_KiemTra_DuocCapNhatCongNo] @NgayGhiNo, @UserName

	IF (@DuocCapNhat = 0)
	BEGIN
	   INSERT INTO [dbo].[CONGNO]
			  ([NoiDung],[NgayGhiNo],[DR],[CR],[UserName]
			   ,[GhiChu],[Status],[style], LoHangID, NguoiTao)
			VALUES
			(@NoiDung,@NgayGhiNo,@DR,@CR,@UserName
			,@GhiChu,@Status,@LoaiPhatSinh,@LoHangID, @NguoiTao
			)
	END
END;

CREATE PROCEDURE [dbo].[CongNo_Top_DanhSach]
	(
	@UserName nvarchar(100) 
	,@Status int
	)
AS
BEGIN

	if @UserName ='' and @Status = -1 -- chon tat ca
			SELECT top 100 * FROM [CONGNO]  where style =2 
			order by CongNo_ID desc,NgayGhiNo desc
	else if @UserName ='' and @Status <> -1 -- chon tat ca user va chon loai status
		begin
			SELECT top 100 * FROM [CONGNO] 
			where  style =2 and [Status] = @Status 
			order by CongNo_ID desc,NgayGhiNo desc;
		end
	else if @UserName <>'' and @Status =-1 -- chon user & tất ca các status
		begin
			SELECT top 100 * FROM [CONGNO] 
			where style =2 and [UserName] = @UserName 
			order by CongNo_ID desc,NgayGhiNo desc;
		end
	else if @UserName <>'' and @Status <>-1 -- chon user chọn status
		begin
			SELECT top 100 * FROM [CONGNO] 
			where style =2 and [UserName] = @UserName  and [Status] = @Status 
			order by CongNo_ID desc,NgayGhiNo desc;
		end
END;

CREATE PROCEDURE [dbo].[CongNo_Update_Status]
(
	@CongNo_ID int 
	,@Status int
	,@NguoiTao nvarchar(250)
)
AS
BEGIN
	DECLARE @NgayGhiNo datetime
	DECLARE @UserName nvarchar(250)
	SELECT @NgayGhiNo = NgayGhiNo, @UserName = UserName FROM CONGNO WHERE CongNo_ID = @CongNo_ID

	DECLARE	@DuocCapNhat int
	EXEC @DuocCapNhat = [SP_KiemTra_DuocCapNhatCongNo] @NgayGhiNo, @UserName

	IF (@DuocCapNhat = 0)
	BEGIN
		update [CONGNO] 
		set [Status] = @Status
		, NgayCapNhatCuoi = GETDATE()
		, NguoiCapNhatCuoi = @NguoiTao
		where [CongNo_ID] = @CongNo_ID 
	END
END;

CREATE PROCEDURE [dbo].[DonHang_InsertItem]
	@username nvarchar(100)
	,@usernamesave nvarchar(100)
	,@linkweb nvarchar(500)
	,@linkhinh nvarchar(500)
	,@corlor nvarchar(50)
	,@size nvarchar(50)
	,@soluong int
	,@dongiaweb float
	,@saleoff float
	,@phuthu float
	,@shipUSA float
	,@tax float
	,@cong float
	,@loaitien nvarchar(10)
	,@ghichu nvarchar(max)
	,@tygia float
	,@giasauoffUSD float
	,@giasauoffVND float
	,@tiencongUSD float
	,@tiencongVND float
	,@tongtienUSD float
	,@tongtienVND float
	,@ordernumber nvarchar(50)
	,@trangthaiOrder nvarchar(50)
	,@ngaysaveLink date
	,@ngaymuahang date
AS
BEGIN
	
	DECLARE	@NgayGhiNo datetime
	SET @NgayGhiNo = GETDATE();

	DECLARE	@DuocCapNhat int
	EXEC @DuocCapNhat = [SP_KiemTra_DuocCapNhatCongNo] @NgayGhiNo, @username

	IF (@DuocCapNhat = 0)
	BEGIN
		INSERT INTO [dbo].[DON_HANG] ([username] ,[usernamesave]  ,[linkweb] ,[linkhinh] ,[corlor]
           ,[size] ,[soluong] ,[dongiaweb] ,[saleoff] ,[phuthu]
           ,[shipUSA] ,[tax] ,[cong] ,[loaitien] ,[ghichu]
           ,[tygia] ,[giasauoffUSD] ,[giasauoffVND] ,[tiencongUSD] ,[tiencongVND]
           ,[tongtienUSD] ,[tongtienVND] ,[ordernumber] ,[trangthaiOrder] ,[ngaysaveLink]
           ,[ngaymuahang] )
		   values ( @username ,@usernamesave ,@linkweb ,@linkhinh ,@corlor
			,@size ,@soluong  ,@dongiaweb ,@saleoff  ,@phuthu  
			,@shipUSA  ,@tax  ,@cong  ,@loaitien ,@ghichu
			,@tygia ,@giasauoffUSD ,@giasauoffVND ,@tiencongUSD ,@tiencongVND
			,@tongtienUSD ,@tongtienVND ,@ordernumber ,@trangthaiOrder ,@ngaysaveLink
			,@ngaymuahang )

		DECLARE @ID int;
		SET @ID = @@IDENTITY;
	
		IF(
			(@trangthaiOrder = 'Ordered' OR @trangthaiOrder = 'Shipped' OR @trangthaiOrder = 'Completed') 
			 AND (@ordernumber is not null AND @ordernumber <> '')
			)
		BEGIN
			-- insert CONGNO
			INSERT INTO CONGNO(DonHang_ID, NoiDung, NgayGhiNo, DR, CR, UserName, [Status], style, NguoiTao)
			VALUES (@ID, N'Phí mua ' + CAST(@ID as nvarchar(20)), @NgayGhiNo, CAST(CEILING(@tongtienVND * 1) / 1 as int), 0, @username, 1, 1, @usernamesave)
		END
		ELSE IF( -- Nếu đang ở trạng thái Ordered/Completed mà chuyển qua Received/Cancelled thì xóa khỏi CONGNO
			(@trangthaiOrder = 'Received' OR @trangthaiOrder = 'Cancelled')
		)
		BEGIN
			DELETE CONGNO WHERE DonHang_ID = @ID
		END
	END
END;

CREATE PROCEDURE [dbo].[SP_Approve_CongNo]
	@CongNo_ID int
	,@NguoiTao nvarchar(250)
AS
BEGIN
	DECLARE @NgayGhiNo datetime
	DECLARE @UserName nvarchar(250)
	SELECT @NgayGhiNo = NgayGhiNo, @UserName = UserName FROM CONGNO WHERE CongNo_ID = @CongNo_ID

	DECLARE	@DuocCapNhat int
	EXEC @DuocCapNhat = [SP_KiemTra_DuocCapNhatCongNo] @NgayGhiNo, @UserName

	IF (@DuocCapNhat = 0)
	BEGIN
		UPDATE CONGNO 
		SET [Status] = 1 
		, NgayCapNhatCuoi = GETDATE()
		, NguoiCapNhatCuoi = @NguoiTao
		WHERE CongNo_ID = @CongNo_ID
	END
END;

CREATE PROCEDURE [dbo].[SP_BaoCao_ChiTietCongNo]
	@username nvarchar(100)
	,@PageSize int
	,@PageNum int
AS
BEGIN
	SET QUERY_GOVERNOR_COST_LIMIT 0;

	SELECT count(*) AS TOTALROW
	FROM [CONGNO] CN
	WHERE CN.DaXoa = 0 and CN.[Status] =1 and CN.[UserName] = @username;

	WITH cte_CongNos AS (
    SELECT 
			ROW_NUMBER() OVER(ORDER BY [CongNo_ID] desc) [STT]
			,[CongNo_ID]
		FROM [CONGNO]
		WHERE DaXoa = 0 and [Status] =1 and [UserName] = @username
	) SELECT 
		CN.[CongNo_ID], CN.[NgayGhiNo], CN.[NoiDung], CN.[DR], CN.[CR], LK.LuyKe, CN.GhiChu
	FROM 
		cte_CongNos CTE
		INNER JOIN [CONGNO] CN ON CTE.[CongNo_ID] = CN.[CongNo_ID]
		INNER JOIN (select t1.CongNo_ID, t1.CanDoi, SUM(t2.CanDoi) as LuyKe
						from CONGNO t1
						inner join CONGNO t2 on t1.CongNo_ID >= t2.CongNo_ID
						where t1.DaXoa = 0 and t2.DaXoa = 0 and t1.[UserName] = @username and t2.[UserName] = @username and t1.[Status] =1 and t2.[Status] =1
						group by t1.CongNo_ID, t1.CanDoi
						) LK ON CN.CongNo_ID = LK.CongNo_ID
	WHERE STT between (@PageNum-1) * @PageSize+1 and (@PageNum-1) * @PageSize + @PageSize
	ORDER BY STT

	SELECT 'A' as Loai, sum(DR) as SoTien
	FROM [CONGNO] CN
	left join Don_Hang DH on CN.DonHang_ID = DH.ID
	where CN.DaXoa = 0 and CN.username = @username and (trangthaiorder in ('Ordered', 'Completed', 'Shipped') or (trangthaiorder is null)) and [Status] = 1
	union
	SELECT 'B' as Loai, sum(CR) as SoTien
	FROM [CONGNO] CN
	where CN.DaXoa = 0 and CN.username = @username and [Status] = 1
	union
	SELECT 'C' as Loai, sum(DR) as SoTien
	FROM [CONGNO] CN
	left join Don_Hang DH on CN.DonHang_ID = DH.ID
	where CN.DaXoa = 0 and CN.username = @username and trangthaiorder = 'Completed' and [Status] = 1
	union
	SELECT 'F' as Loai, sum(DR) as SoTien
	FROM [CONGNO] CN
	left join Don_Hang DH on CN.DonHang_ID = DH.ID
	where CN.DaXoa = 0 and CN.username = @username and trangthaiorder = 'Ordered' and [Status] = 1

END;

			FROM tbKy Ky
			LEFT JOIN tbChotKy CK on Ky.KyID = CK.KyID and UserName = @username

	--			group by t1.CongNo_ID, t1.STT

CREATE PROCEDURE [dbo].[SP_BaoCao_CongNoKhachHang]
AS
BEGIN
	select CDCN.UserName, CDCN.TienNo, CDCN.TienHangChuaGiao
	, CASE WHEN CDCN.TienHangChuaGiao = 0 AND CDCN.TienNo > 0 THEN 999999 
			WHEN CDCN.TienHangChuaGiao = 0 AND CDCN.TienNo < 0 THEN -999999 
			WHEN CDCN.TienHangChuaGiao = 0 AND CDCN.TienNo = 0 THEN 0 
			ELSE CDCN.TienNo*100/CDCN.TienHangChuaGiao END as PhanTram
	 from
	(
		SELECT CN.[UserName]
			, sum([CanDoi]) as TienNo
			, Sum(CASE WHEN DH.trangthaiorder = 'Ordered' THEN CN.DR ELSE 0 END) AS TienHangChuaGiao
			--, sum([CanDoi]) / Sum(CASE WHEN DH.trangthaiorder = 'Ordered' THEN CN.DR ELSE 0 END)
		FROM [CONGNO] CN
		left join Don_Hang DH on CN.DonHang_ID = DH.ID and trangthaiorder = 'Ordered'
		WHERE CN.HienTai = 1 and CN.DaXoa = 0 and [Status] = 1
		GROUP BY CN.[UserName]
	) CDCN
	ORDER BY CDCN.TienNo desc
END;

CREATE PROCEDURE [dbo].[SP_BaoCao_CongNoKhachHangTheoLo]
	@TuNgay datetime
	,@DenNgay datetime
AS
BEGIN
	SELECT KQ.*
	, (KQ.TienLoHangA + KQ.TienPhiHaiQuanB) as TongTienLoHangAB
	, ((KQ.TienLoHangA + KQ.TienPhiHaiQuanB) - DaThu) as ConLai
	FROM   
	(
		SELECT LH.*
		, ISNULL((select sum(TongTienShipVeVN_VND) from tbLoHang_PhiShipVeVN where LoHangID = LH.LoHangID),0) as TienLoHangA
		, ISNULL((select sum(TongTienThueHaiQuan_VND) from tbLoHang_ThueHaiQuan where LoHangID = LH.LoHangID),0) as TienPhiHaiQuanB
		--, ISNULL((select sum(TienVND) from tbLoHang_ChiPhiLoHang where LoHangID = LH.LoHangID),0) as TienChiPhiLoHangC
		, ISNULL((select sum(CR) from CONGNO where HienTai = 1 and DaXoa = 0 and LoHangID = LH.LoHangID),0) as DaThu
		FROM tbLoHang LH
		WHERE NgayLoHang between @TuNgay and @DenNgay
	) KQ
	ORDER BY KQ.NgayLoHang desc, KQ.LoHangID desc

END;

CREATE PROCEDURE [dbo].[SP_BaoCao_CongNoTheoDotHang]
	@TuNgay datetime
	,@DenNgay datetime
AS
BEGIN
	SELECT KQ.*, U.DiaChi, U.PhoneNumber, U.HoTen, DH.DiaChiNhanHang
	FROM 
	(
		SELECT CN1.username
				, CN1.TenDotHang
				, CN1.ngayveVN
				, sum(DR) as TongTien
				, sum(CASE WHEN [style] in (3,4) THEN DR ELSE 0 END) as TienShip
				, sum(CASE WHEN [style] = 1 THEN DR ELSE 0 END) as TienHang 
				, sum(CASE WHEN [style] = 3 THEN DR ELSE 0 END) as TienShipVeVN
				, sum(CASE WHEN [style] = 4 THEN DR ELSE 0 END) as TienShipTrongNuoc
				, sum(CASE WHEN [style] = 5 THEN CR ELSE 0 END) as TienDatCoc
		FROM
			(
				SELECT CN.UserName, CN.DR, CN.CR, CN.style, DH.TenDotHang, DH.ngayveVN
				FROM CONGNO CN 
				INNER JOIN DON_HANG DH on CN.DonHang_ID = DH.ID
				WHERE CN.DaXoa = 0
				and CN.HienTai = 1
				and CN.[Status] = 1
				and CN.[style] in (1,3,4,5)
				and DH.DaXoa = 0
				and DH.TenDotHang in (select distinct TenDotHang from DON_HANG where ngayveVN between @TuNgay and @DenNgay)
				and (DH.trangthaiorder in ('Ordered', 'Completed', 'Shipped') or (DH.trangthaiorder is null))
		
				UNION ALL
		
				SELECT CN.UserName, CN.DR, CN.CR, CN.style, CN.TenDotHang, CONVERT(datetime,  CN.TenDotHang, 112)
				FROM CONGNO CN 
				WHERE CN.DaXoa = 0
				and CN.HienTai = 1
				and CN.[Status] = 1
				and CN.[style] in (1,3,4,5)
				and CN.TenDotHang in (select distinct TenDotHang from DON_HANG where ngayveVN between @TuNgay and @DenNgay)
			) CN1
		GROUP BY CN1.username, CN1.TenDotHang, CN1.ngayveVN
	) KQ
	INNER JOIN AspNetUsers U ON KQ.username = U.UserName
	LEFT JOIN DotHang DH ON KQ.TenDotHang = DH.TenDotHang and KQ.username = DH.UserName
	ORDER BY KQ.TenDotHang desc, KQ.UserName


	--SELECT username, sum(tongtienVND) as tongtienVND, sum(CanHang_TienShipVeVN) as CanHang_TienShipVeVN, sum(CanHang_TienShipTrongNuoc) as CanHang_TienShipTrongNuoc FROM DON_HANG WHERE TenDotHang = @TenDotHang and DaXoa = 0 and (trangthaiorder in ('Ordered', 'Completed', 'Shipped') or (trangthaiorder is null)) group by username
END;

CREATE PROCEDURE [dbo].[SP_BaoCao_DoiChieuCongNo]
	@TuNgay datetime
	,@OrderNumber nvarchar(100)
AS
BEGIN
	CREATE TABLE #tbTemp (ordernumber nvarchar(100))

	IF(@OrderNumber <> '')
	BEGIN
		SET @OrderNumber = '%' + @OrderNumber + '%'
		INSERT INTO #tbTemp
			select distinct ordernumber from DON_HANG where DaXoa = 0 and trangthaiOrder in ('Ordered', 'Shipped', 'Completed') and  ngaymuahang >= @TuNgay and ordernumber like @OrderNumber
	END
	ELSE
	BEGIN
		INSERT INTO #tbTemp
			select distinct ordernumber from DON_HANG where DaXoa = 0 and trangthaiOrder in ('Ordered', 'Shipped', 'Completed') and  ngaymuahang >= @TuNgay
	END
	
	SELECT A.*, B.SoLinkB, B.SotienB, B.SotienBVND, B.tracking_number, KiemTra = CASE WHEN A.SoTienA < B.SotienB AND abs(A.SoTienA - B.SotienB) >= 0.1 THEN 'Sai' ELSE '' END, KiemTraVND = CASE WHEN A.SoTienAVND < B.SotienBVND AND abs(A.SoTienAVND - B.SotienBVND) >= 1000 THEN 'Sai' ELSE '' END
	FROM (select DH.ordernumber, SUM(soluong) as SoLinkA, ISNULL(SUM(tongtienUSD), 0) as SoTienA, ISNULL(SUM(tongtienVND), 0) as SoTienAVND, max([ngaymuahang]) as ngaymuahang
			from DON_HANG DH
			inner join #tbTemp T on DH.ordernumber = T.ordernumber
			where DaXoa = 0 
			group by DH.ordernumber
			) A
	LEFT JOIN (select LO.OrderNumber,SUM(TongSoLink) as SoLinkB, ISNULL(SUM(TongTienOrder), 0) as SotienB, ISNULL(SUM(TongTienOrderVND), 0) as SotienBVND, tracking_number
				from LIST_ORDER LO
				inner join #tbTemp T on LO.ordernumber = T.ordernumber
				group by LO.OrderNumber, tracking_number
				) B ON A.ordernumber = B.OrderNumber
	ORDER BY A.ngaymuahang desc
	
	-- quanglm them ngay 28/09/2016
	SELECT ISNULL(SUM(A.SoTienA), 0) as SoTienA
	 ,ISNULL(SUM(A.SoTienAVND), 0) as SoTienAVND
	 ,ISNULL(SUM(B.SotienB), 0) as SoTienB 
	 ,ISNULL(SUM(B.SotienBVND), 0) as SoTienBVND 
	 ,ISNULL(SUM(A.SoTienCong), 0) as SoTienCong
	 ,ISNULL(SUM(A.SoTienCongVND), 0) as SoTienCongVND
	 ,ISNULL(SUM(A.SoTienA), 0) - ISNULL(SUM(B.SotienB), 0) as TienChenhLech 
	 ,ISNULL(SUM(A.SoTienAVND), 0) - ISNULL(SUM(B.SotienBVND), 0) as TienChenhLechVND 
	FROM (select DH.ordernumber, SUM(soluong) as SoLinkA, ISNULL(SUM(tongtienUSD), 0) as SoTienA, ISNULL(SUM(tongtienVND), 0) as SoTienAVND, max([ngaymuahang]) as ngaymuahang,ISNULL(SUM(tiencongUSD), 0) as SoTienCong,ISNULL(SUM(tiencongVND), 0) as SoTienCongVND
			from DON_HANG DH
			inner join #tbTemp T on DH.ordernumber = T.ordernumber
			where DaXoa = 0 
			group by DH.ordernumber
			) A
	LEFT JOIN (select LO.OrderNumber,SUM(TongSoLink) as SoLinkB, ISNULL(SUM(TongTienOrder), 0) as SotienB, ISNULL(SUM(TongTienOrderVND), 0) as SotienBVND, tracking_number
				from LIST_ORDER LO
				inner join #tbTemp T on LO.ordernumber = T.ordernumber
				group by LO.OrderNumber, tracking_number
				) B ON A.ordernumber = B.OrderNumber

	DROP TABLE #tbTemp;
END;

CREATE PROCEDURE [dbo].[SP_BaoCao_InPhieuShipTheoDotHang]
	@TenDotHang nvarchar(50)
	,@UserName nvarchar(250)
AS
BEGIN
	SELECT KQ.*, U.DiaChi, U.PhoneNumber, U.HoTen, DH.DiaChiNhanHang, DH.SoVanDon, DH.NgayGuiHang
	FROM 
	(
		SELECT CN1.username
				, CN1.TenDotHang
				, CN1.ngayveVN
				, sum(DR) as TongTien
				, sum(CASE WHEN [style] in (3,4) THEN DR ELSE 0 END) as TienShip
				, sum(CASE WHEN [style] = 1 THEN DR ELSE 0 END) as TienHang 
				, sum(CASE WHEN [style] = 3 THEN DR ELSE 0 END) as TienShipVeVN
				, sum(CASE WHEN [style] = 4 THEN DR ELSE 0 END) as TienShipTrongNuoc
				, sum(CASE WHEN [style] = 5 THEN CR ELSE 0 END) as TienDatCoc
		FROM
			(
				SELECT CN.UserName, CN.DR, CN.CR, CN.style, DH.TenDotHang, DH.ngayveVN
				FROM CONGNO CN 
				INNER JOIN DON_HANG DH on CN.DonHang_ID = DH.ID
				WHERE CN.DaXoa = 0
				and CN.HienTai = 1
				and CN.[Status] = 1
				and CN.[style] in (1,3,4,5)
				and DH.DaXoa = 0
				and DH.TenDotHang = @TenDotHang
				and CN.UserName = @UserName
				and (DH.trangthaiorder in ('Ordered', 'Completed', 'Shipped') or (DH.trangthaiorder is null))
		
				UNION ALL
		
				SELECT CN.UserName, CN.DR, CN.CR, CN.style, CN.TenDotHang, CONVERT(datetime,  CN.TenDotHang, 112)
				FROM CONGNO CN 
				WHERE CN.DaXoa = 0
				and CN.HienTai = 1
				and CN.[Status] = 1
				and CN.[style] in (1,3,4,5)
				and CN.TenDotHang = @TenDotHang
				and CN.UserName = @UserName
			) CN1
		GROUP BY CN1.username, CN1.TenDotHang, CN1.ngayveVN
	) KQ
	INNER JOIN AspNetUsers U ON KQ.username = U.UserName
	LEFT JOIN DotHang DH ON KQ.TenDotHang = DH.TenDotHang and KQ.username = DH.UserName
	ORDER BY KQ.TenDotHang desc, KQ.UserName

	SELECT DH.*, HM.DaQuaHanMuc, HM.LaKhachVip, LH.TenLoaiHang
	FROM Don_Hang DH
	LEFT JOIN HanMucKhachHang HM ON DH.username = HM.UserName
	LEFT JOIN LoaiHang LH ON DH.LoaiHangID = LH.LoaiHangID 
	WHERE DH.TenDotHang = @TenDotHang and DH.username = @UserName and DH.DaXoa = 0 and (DH.trangthaiorder in ('Ordered', 'Completed', 'Shipped') or (DH.trangthaiorder is null))
	ORDER BY DH.ngaymuahang desc
END;

CREATE PROCEDURE [dbo].[SP_BaoCao_LaiLoTheoLo]
	@TuNgay datetime
	,@DenNgay datetime
AS
BEGIN
	SELECT KQ.*
	, (KQ.TienLoHangA + KQ.TienPhiHaiQuanB) as TongTienLoHangAB
	, ((KQ.TienLoHangA + KQ.TienPhiHaiQuanB) - TienChiPhiLoHangC) as LaiLoD
	FROM   
	(
		SELECT LH.NgayLoHang, LH.TenLoHang
		, ISNULL((select sum(TongTienShipVeVN_VND) from tbLoHang_PhiShipVeVN where LoHangID in (select LoHangID from tbLoHang where TenLoHang = LH.TenLoHang)),0) as TienLoHangA
		, ISNULL((select sum(TongTienThueHaiQuan_VND) from tbLoHang_ThueHaiQuan where LoHangID in (select LoHangID from tbLoHang where TenLoHang = LH.TenLoHang)),0) as TienPhiHaiQuanB
		, ISNULL((select sum(TienVND) from tbLoHang_ChiPhiLoHang where TenLoHang = LH.TenLoHang),0) as TienChiPhiLoHangC
		FROM tbLoHang LH
		WHERE NgayLoHang between @TuNgay and @DenNgay
		GROUP BY NgayLoHang, TenLoHang
	) KQ
	ORDER BY KQ.NgayLoHang desc

END;

CREATE PROCEDURE [dbo].[SP_CanHang]
	@ID int
	,@LoaiHangID int
	,@CanHang_SoKg float
	,@CanHang_TienShipVeVN float
	,@CanHang_TienShipTrongNuoc float
	,@CanHang_GhiChuShipVeVN nvarchar(max)
	,@NguoiTao nvarchar(250)
AS
BEGIN TRY
	SET NOCOUNT ON;
	BEGIN TRANSACTION

	DECLARE @username nvarchar(50)
	SELECT @username = username FROM [DON_HANG] WHERE ID =@ID

	DECLARE	@NgayGhiNo datetime
	SET @NgayGhiNo = GETDATE();
	IF(EXISTS(SELECT * FROM CONGNO WHERE HienTai = 1 AND DonHang_ID = @ID AND style = 3))
	BEGIN
		SET @NgayGhiNo = (SELECT top 1 NgayGhiNo FROM CONGNO WHERE HienTai = 1 AND DonHang_ID = @ID AND style = 3)
	END

	DECLARE	@DuocCapNhat int
	EXEC @DuocCapNhat = [SP_KiemTra_DuocCapNhatCongNo] @NgayGhiNo, @username

	IF (@DuocCapNhat = 0)
	BEGIN
		UPDATE [dbo].[DON_HANG]
		SET LoaiHangID = @LoaiHangID
			,CanHang_SoKg = @CanHang_SoKg
			,CanHang_TienShipVeVN = @CanHang_TienShipVeVN
			,CanHang_TienShipTrongNuoc = @CanHang_TienShipTrongNuoc
			,trangthaiOrder = 'Shipped'
		WHERE ID =@ID

		--Phí ship về VN
		IF(EXISTS(SELECT * FROM CONGNO WHERE HienTai = 1 AND DonHang_ID = @ID AND style = 3))
		BEGIN
			UPDATE CONGNO
			SET DR = CAST(CEILING(@CanHang_TienShipVeVN * 1) / 1 as int)
			, UserName = @username
			, DaXoa = 0
			, GhiChu = 'CN: ' + CAST(@CanHang_SoKg as nvarchar(50)) + ' - ' + @CanHang_GhiChuShipVeVN
			, NgayCapNhatCuoi = GETDATE()
			, NguoiCapNhatCuoi = @NguoiTao
			WHERE HienTai = 1 AND DonHang_ID = @ID AND style = 3
		END
		ELSE
		BEGIN
			INSERT INTO CONGNO(DonHang_ID, NoiDung, NgayGhiNo, DR, CR, UserName, [Status], style, GhiChu, HienTai, NguoiTao)
			VALUES (@ID, N'Phí ship về VN ' + CAST(@ID as nvarchar(20)), @NgayGhiNo, CAST(CEILING(@CanHang_TienShipVeVN * 1) / 1 as int), 0, @username, 1, 3, 'CN: ' + CAST(@CanHang_SoKg as nvarchar(50)) + ' - ' + @CanHang_GhiChuShipVeVN, 1, @NguoiTao)
		END
	END

COMMIT TRANSACTION
END TRY
BEGIN CATCH
	-- Whoops, there was an error
	 IF @@TRANCOUNT > 0
	 ROLLBACK TRANSACTION
	-- Raise an error with the details of the exception
	 DECLARE @ErrMsg nvarchar(4000), @ErrSeverity int

	 SELECT @ErrMsg = ERROR_MESSAGE(),
	 @ErrSeverity = ERROR_SEVERITY()

	 RAISERROR(@ErrMsg, @ErrSeverity, 1)
END CATCH;

CREATE PROCEDURE [dbo].[SP_CapNhat_CongNo]
	@CongNo_ID int
	,@NoiDung nvarchar(100)
	--,@NgayGhiNo datetime
	,@DR float
	,@CR float
	,@UserName nvarchar(100)
	,@GhiChu nvarchar(500)
	,@Status int
	,@LoHangID int
	,@NguoiTao nvarchar(250)
	,@udNoiDung bit
	,@udDR bit
	,@udCR bit
	,@udUserName bit
	,@udGhiChu bit
	,@udStatus bit
	,@udLoHangID bit
AS
	
BEGIN
	
	DECLARE @usernameOld nvarchar(50)
	DECLARE	@NgayGhiNoOld datetime
	SELECT @usernameOld = username, @NgayGhiNoOld = NgayGhiNo FROM CONGNO WHERE CongNo_ID = @CongNo_ID

	DECLARE	@DuocCapNhat int
	DECLARE	@DuocCapNhatOld int
	EXEC @DuocCapNhat = [SP_KiemTra_DuocCapNhatCongNo] @NgayGhiNoOld, @username
	EXEC @DuocCapNhatOld = [SP_KiemTra_DuocCapNhatCongNo] @NgayGhiNoOld, @usernameOld

	IF ((@DuocCapNhat = 0) AND (@DuocCapNhatOld = 0))
	BEGIN
		UPDATE CONGNO
		SET 
		--,[NgayGhiNo] = @NgayGhiNo
		NoiDung = (CASE WHEN @udNoiDung = 1 THEN @NoiDung ELSE NoiDung END)
		,[DR] = (CASE WHEN @udDR = 1 THEN @DR ELSE [DR] END)
		,[CR] = (CASE WHEN @udCR = 1 THEN @CR ELSE [CR] END)

		,[GhiChu] = (CASE WHEN @udGhiChu = 1 THEN @GhiChu ELSE [GhiChu] END)
		,[Status] = (CASE WHEN @udStatus = 1 THEN @Status ELSE [Status] END)
		,[LoHangID] = (CASE WHEN @udLoHangID = 1 THEN @LoHangID ELSE [LoHangID] END)
		, NgayCapNhatCuoi = GETDATE()
		, NguoiCapNhatCuoi = @NguoiTao
		WHERE CongNo_ID = @CongNo_ID
	END
END;

CREATE PROCEDURE [dbo].[SP_CapNhat_CongNo_LoHang_PhiShipVeVN]
	@LoHangID int
	,@NguoiTao nvarchar(250)
AS
BEGIN
	DECLARE @UserName as nvarchar(250);
	DECLARE @TinhTrang as nvarchar(50);

	SELECT @UserName = UserName FROM tbLoHang where LoHangID = @LoHangID
	
	DECLARE	@NgayGhiNo datetime
	SET @NgayGhiNo = GETDATE();
	IF(EXISTS(SELECT * FROM CONGNO WHERE HienTai = 1 AND LoHangID = @LoHangID AND style = 6))
	BEGIN
		SET @NgayGhiNo = (SELECT top 1 NgayGhiNo FROM CONGNO WHERE HienTai = 1 AND LoHangID = @LoHangID AND style = 6)
	END

	DECLARE	@DuocCapNhat int
	EXEC @DuocCapNhat = [SP_KiemTra_DuocCapNhatCongNo] @NgayGhiNo, @username

	IF (@DuocCapNhat = 0)
	BEGIN
		DECLARE @TongTienShipVeVN int
		SELECT @TongTienShipVeVN = SUM(TongTienShipVeVN_VND) FROM tbLoHang_PhiShipVeVN WHERE LoHangID = @LoHangID

		--Phí ship về VN
		IF(@TongTienShipVeVN > 0)
		BEGIN
			IF(EXISTS(SELECT * FROM CONGNO WHERE HienTai = 1 AND LoHangID = @LoHangID AND style = 6))
			BEGIN
				UPDATE CONGNO
				SET DR = CAST(CEILING(@TongTienShipVeVN * 1) / 1 as int)
				, UserName = @UserName
				, DaXoa = 0
				, NgayCapNhatCuoi = GETDATE()
				, NguoiCapNhatCuoi = @NguoiTao
				WHERE HienTai = 1 AND LoHangID = @LoHangID AND style = 6
			END
			ELSE
			BEGIN
				INSERT INTO CONGNO(LoHangID, NoiDung, NgayGhiNo, DR, CR, UserName, [Status], style, HienTai, NguoiTao)
				VALUES (@LoHangID, N'Phí ship về VN lô hàng ' + CAST(@LoHangID as nvarchar(20)), GETDATE(), CAST(CEILING(@TongTienShipVeVN * 1) / 1 as int), 0, @UserName, 1, 6, 1, @NguoiTao)
			END
		END
	END
END;

CREATE PROCEDURE [dbo].[SP_CapNhat_LoHang]
	@LoHangID int
	,@UserName nvarchar(250)
	,@LoaiTien nvarchar(50)
	,@TyGia int
	,@NgayDenDuKien datetime
	,@NgayDenThucTe datetime
	,@NguoiTao nvarchar(250)
AS
BEGIN TRY
	SET NOCOUNT ON;
	BEGIN TRANSACTION
	DECLARE @TyGia_HienTai nvarchar(50)
	SELECT @TyGia_HienTai = TyGia FROM [tbLoHang] WHERE LoHangID = @LoHangID

	UPDATE [dbo].[tbLoHang]
	SET LoaiTien = @LoaiTien

		--,UserName = @UserName
		,NgayDenDuKien = @NgayDenDuKien
		,NgayDenThucTe = @NgayDenThucTe
		,NguoiTao = @NguoiTao
	WHERE LoHangID =@LoHangID

	IF(@TyGia_HienTai <> @TyGia)
	BEGIN
		UPDATE tbLoHang_PhiShipVeVN
		SET TongTienShipVeVN_VND = CAST(CEILING(CanNang * DonGia * @TyGia * 1) / 1 as int)
		WHERE LoHangID = @LoHangID

		EXEC SP_CapNhat_CongNo_LoHang_PhiShipVeVN @LoHangID


		UPDATE tbLoHang_ThueHaiQuan
		SET TongTienThueHaiQuan_VND = CAST(CEILING(CanNangSoLuongGiaTri * DonGia * @TyGia * 1 ) / 1 as int)
		WHERE LoHangID = @LoHangID

		EXEC SP_CapNhat_CongNo_LoHang_ThueHaiQuan @LoHangID
	END

COMMIT TRANSACTION
END TRY
BEGIN CATCH
	-- Whoops, there was an error
	 IF @@TRANCOUNT > 0
	 ROLLBACK TRANSACTION
	-- Raise an error with the details of the exception
	 DECLARE @ErrMsg nvarchar(4000), @ErrSeverity int

	 SELECT @ErrMsg = ERROR_MESSAGE(),
	 @ErrSeverity = ERROR_SEVERITY()

	 RAISERROR(@ErrMsg, @ErrSeverity, 1)
END CATCH;

		, NgayCapNhatCuoi = GETDATE()

CREATE PROCEDURE [dbo].[SP_CapNhat_MassCompleteAllTrackingWithFilter]
	@UserName nvarchar(250)
	,@TinhTrangFilter nvarchar(100)
	,@NoiDungTim nvarchar(250)
	,@TimTheo int
	,@TrackingNumber nvarchar(500)
	,@TenLoHang nvarchar(50)
	,@QuocGiaID int
	,@DaXoa bit
	,@TuNgay nvarchar(15)
	,@DenNgay nvarchar(15)
	,@TinhTrang nvarchar(50)
	,@NgayChuyenTinhTrang datetime
	,@NguoiTao nvarchar(250)
AS
BEGIN
	DECLARE @SqlWhere nvarchar(MAX);
	SET @SqlWhere = ' WHERE DaXoa = ' + CAST(@DaXoa AS CHAR(1))
	IF(@UserName <> '')
	BEGIN
		SET @SqlWhere = @SqlWhere + ' AND UserName = ''' + @UserName + ''''
	END
	
	IF(@TinhTrang <> '')
	BEGIN
		SET @SqlWhere = @SqlWhere + ' AND TinhTrang in (' + @TinhTrangFilter + ')'
	END
	
	IF(@NoiDungTim <> '')
	BEGIN
		IF(@TimTheo = -1)
		BEGIN
			SET @SqlWhere = @SqlWhere + ' AND ((TrackingID like N''%' + @NoiDungTim + '%'') OR (OrderNumber like N''%' + @NoiDungTim + '%'') OR (TenLoHang like N''%' + @NoiDungTim + '%'') OR (TrackingNumber like N''%' + @NoiDungTim + '%'') OR (GhiChu like N''%' + @NoiDungTim + '%'') OR (Kien like N''%' + @NoiDungTim + '%'') OR (Mawb like N''%' + @NoiDungTim + '%'') OR (Hawb like N''%' + @NoiDungTim + '%'') OR (TenQuocGia like N''%' + @NoiDungTim + '%''))'
		END
		ELSE IF(@TimTheo = 0)
		BEGIN
			SET @SqlWhere = @SqlWhere + ' AND (OrderNumber like N''%' + @NoiDungTim + '%'')'
		END
	END
	
	IF(@TrackingNumber <> '')
	BEGIN
		SET @SqlWhere = @SqlWhere + ' AND TrackingNumber like ''%' + @TrackingNumber + '%'''
	END

	IF(@TenLoHang <> '')
	BEGIN
		SET @SqlWhere = @SqlWhere + ' AND TenLoHang = ''' + @TenLoHang + ''''
	END

	IF(@QuocGiaID <> -1)
	BEGIN
		SET @SqlWhere = @SqlWhere + ' AND TK.QuocGiaID = ' + CAST(@QuocGiaID as nvarchar(10))
	END

	IF(@TuNgay <> '')
	BEGIN
		SET @SqlWhere = @SqlWhere + ' AND NgayDatHang >= ''' + @TuNgay + ''''
	END

	IF(@DenNgay <> '')
	BEGIN
		SET @SqlWhere = @SqlWhere + ' AND NgayDatHang <= ''' + @DenNgay+ ''''
	END


	--Update tracking
	DECLARE @SqlUpdateTracking nvarchar(MAX);
	SET @SqlUpdateTracking = 'UPDATE tbTracking SET TinhTrang = ''Completed'' ' + @SqlWhere
	EXEC sp_executesql @SqlUpdateTracking;

	IF @NgayChuyenTinhTrang is null
	BEGIN
		SET @NgayChuyenTinhTrang = getdate()
	END

	--Update insert tinhtrangtracking
	DECLARE @SqlInsertTinhTrangTracking nvarchar(MAX);
	SET @SqlInsertTinhTrangTracking = 'INSERT INTO tbTinhTrangTracking (TrackingID, TinhTrang, NgayChuyenTinhTrang, NgayTao, NguoiTao)'
	SET @SqlInsertTinhTrangTracking = @SqlInsertTinhTrangTracking + ' SELECT TrackingID, ''Completed'', ''' + convert(varchar, @NgayChuyenTinhTrang, 120) +''', '''+ convert(varchar, getdate(), 120) +''', ''' + @NguoiTao + ''''
	SET @SqlInsertTinhTrangTracking = @SqlInsertTinhTrangTracking + ' FROM tbTracking ' + @SqlWhere
	EXEC sp_executesql @SqlUpdateTracking;
	
	EXEC sp_executesql @SqlInsertTinhTrangTracking;

END;

CREATE PROCEDURE [dbo].[SP_CapNhat_NgayLoHang]
	@id nvarchar(4000)
	,@NgayLoHang datetime
	,@GhiChu nvarchar(500)
AS
BEGIN TRY
	SET NOCOUNT ON;
	BEGIN TRANSACTION
	
	DECLARE @TenLoHang nvarchar(500)
	DECLARE @UserName nvarchar(100)

	SET @TenLoHang = (CONVERT([nvarchar](50),@NgayLoHang,(112)) + case when @GhiChu='' OR @GhiChu IS NULL then '' else '-' + @GhiChu end)


	UPDATE tbTracking
	SET NgayLoHang = @NgayLoHang 
	, TenLoHang = @TenLoHang
	WHERE TrackingID in (SELECT s as ID FROM dbo.SplitString(@id, ','))

	DECLARE RS SCROLL CURSOR FOR 
	SELECT distinct UserName FROM tbTracking WHERE DaXoa = 0 AND TrackingID in (SELECT s as ID FROM dbo.SplitString(@id, ','))
	
	OPEN RS
	FETCH NEXT FROM RS INTO @UserName

	WHILE( @@FETCH_STATUS = 0)
	BEGIN
		IF(EXISTS (SELECT * FROM tbLoHang WHERE TenLoHang = @TenLoHang AND UserName = @UserName))
		BEGIN
			UPDATE tbLoHang SET DaXoa = 0 WHERE TenLoHang = @TenLoHang AND UserName = @UserName
		END
		ELSE
		BEGIN
			INSERT INTO tbLoHang(UserName, NgayLoHang, GhiChu) VALUES(@UserName, @NgayLoHang, @GhiChu)
		END
		FETCH NEXT FROM RS INTO @UserName
	END
	CLOSE RS
	DEALLOCATE RS

COMMIT TRANSACTION
END TRY
BEGIN CATCH
	-- Whoops, there was an error
	 IF @@TRANCOUNT > 0
	 ROLLBACK TRANSACTION
	-- Raise an error with the details of the exception
	 DECLARE @ErrMsg nvarchar(4000), @ErrSeverity int

	 SELECT @ErrMsg = ERROR_MESSAGE(),
	 @ErrSeverity = ERROR_SEVERITY()

	 RAISERROR(@ErrMsg, @ErrSeverity, 1)
END CATCH;

CREATE PROCEDURE [dbo].[SP_CapNhatDonHang]
	@ID int
	,@ordernumber nvarchar(50)
	,@username nvarchar(100)
	,@usernamesave nvarchar(100)
	,@linkweb nvarchar(500)
	,@linkhinh nvarchar(500)
	,@corlor nvarchar(50)
	,@size nvarchar(50)
	,@soluong int
	,@dongiaweb float
	,@saleoff float
	,@phuthu float
	,@shipUSA float
	,@tax float
	,@cong float
	,@loaitien nvarchar(10)
	,@ghichu nvarchar(max)
	,@tygia float
	,@giasauoffUSD float
	,@giasauoffVND float
	,@tiencongUSD float
	,@tiencongVND float
	,@tongtienUSD float
	,@tongtienVND float
	,@trangthaiOrder nvarchar(50)
	,@ngaymuahang date
	,@ngayveVN date
	,@AdminNote nvarchar(max)
	,@LoaiHangID int
	,@QuocGiaID int
	,@NguoiTao nvarchar(250)
AS
BEGIN TRY
	SET NOCOUNT ON;
	BEGIN TRANSACTION

	
	DECLARE	@DuocCapNhat int
	EXEC @DuocCapNhat = [SP_KiemTra_DuocCapNhatDonHang] @ID, @username

	IF (@DuocCapNhat = 0)
	BEGIN
		DECLARE @usernameOld nvarchar(50)
		SELECT @usernameOld = username FROM DON_HANG WHERE  ID = @ID


		UPDATE [dbo].[DON_HANG]
		SET [ordernumber] = @ordernumber
			,[username] = @username
			,[usernamesave] = @usernamesave
			,[linkweb] = @linkweb
			,[linkhinh] = @linkhinh
			,[corlor] = @corlor
			,[size] = @size
			,[soluong] = @soluong
			,[dongiaweb] = @dongiaweb
			,[saleoff] = @saleoff
			,[phuthu] = @phuthu
			,[shipUSA] = @shipUSA
			,[tax] = @tax
			,[cong] = @cong
			,[loaitien] = @loaitien
			,[ghichu] = @ghichu
			,[tygia] = @tygia
			,giasauoffUSD = @giasauoffUSD
			,[tiencongUSD] = @tiencongUSD
			,[tongtienUSD] = @tongtienUSD
			,giasauoffVND = @giasauoffVND
			,[tongtienVND] = @tongtienVND
			,[tiencongVND] = @tiencongVND
			,[trangthaiOrder] = @trangthaiOrder
			,ngaymuahang = @ngaymuahang
			,ngayveVN = @ngayveVN
			,[AdminNote] = @AdminNote
			,[LoaiHangID] = CASE WHEN @LoaiHangID is not null THEN @LoaiHangID ELSE [LoaiHangID] END
			,[QuocGiaID] = CASE WHEN @QuocGiaID is not null THEN @QuocGiaID ELSE [QuocGiaID] END
		WHERE ID =@ID

		IF(@username <> @usernameOld)
		BEGIN
			UPDATE CONGNO
			SET UserName = @username
			, NguoiCapNhatCuoi = @NguoiTao
			, NgayCapNhatCuoi = GETDATE()
			WHERE DonHang_ID = @ID
		END

		IF(@trangthaiOrder = 'Ordered' OR @trangthaiOrder = 'Shipped' OR @trangthaiOrder = 'Completed') 
		BEGIN
			IF(EXISTS(SELECT * FROM CONGNO WHERE HienTai = 1 AND DonHang_ID = @ID AND style = 1))
			BEGIN
				-- update CONGNO
				UPDATE CONGNO
				SET DR = CAST(CEILING(@tongtienVND * 1) / 1 as int)
				, DaXoa = 0
				, NguoiCapNhatCuoi = @NguoiTao
				, NgayCapNhatCuoi = GETDATE()
				WHERE HienTai = 1 AND DonHang_ID = @ID AND style = 1
			END
			ELSE
			BEGIN
				-- insert CONGNO
				INSERT INTO CONGNO(DonHang_ID, NoiDung, NgayGhiNo, DR, CR, UserName, [Status], style, HienTai, NguoiTao)
				VALUES (@ID, N'Phí mua ' + CAST(@ID as nvarchar(20)), GETDATE(), CAST(CEILING(@tongtienVND * 1) / 1 as int), 0, @username, 1, 1, 1, @NguoiTao)
			END
		END
		ELSE IF( -- Nếu đang ở trạng thái Ordered/Completed mà chuyển qua Received/Cancelled thì xóa khỏi CONGNO
			(@trangthaiOrder = 'Received' OR @trangthaiOrder = 'Cancelled')
		)
		BEGIN
			--DELETE CONGNO WHERE DonHang_ID = @ID
			INSERT INTO CONGNO(DonHang_ID, NoiDung, NgayGhiNo, DR, CR, UserName, [Status], style, GhiChu, HienTai, NguoiTao)
			SELECT DonHang_ID, N'Cập nhật đơn hàng ' + CAST(DonHang_ID as nvarchar(20)), GETDATE(), DR * (-1), CR, UserName, [Status], style, '', 0, @NguoiTao
			FROM CONGNO WHERE HienTai = 1 AND DonHang_ID = @ID
	
			UPDATE CONGNO 
			SET HienTai = 0 
			, NguoiCapNhatCuoi = @NguoiTao
			, NgayCapNhatCuoi = GETDATE()
			WHERE HienTai = 1 AND DonHang_ID = @ID
		
		END
	END

COMMIT TRANSACTION
END TRY
BEGIN CATCH
	-- Whoops, there was an error
	 IF @@TRANCOUNT > 0
	 ROLLBACK TRANSACTION
	-- Raise an error with the details of the exception
	 DECLARE @ErrMsg nvarchar(4000), @ErrSeverity int

	 SELECT @ErrMsg = ERROR_MESSAGE(),
	 @ErrSeverity = ERROR_SEVERITY()

	 RAISERROR(@ErrMsg, @ErrSeverity, 1)
END CATCH;

CREATE PROCEDURE [dbo].[SP_DongKy]
	@KyCanDongID int
	, @UserName nvarchar(50)
	, @LaKyDauTien bit
	, @NguoiTao nvarchar(50)
AS
BEGIN
	SET QUERY_GOVERNOR_COST_LIMIT 0;
	--select * from tbky

	declare @IDKyDauTien int;
	SELECT top 1 @IDKyDauTien = KyID FROM tbKy order by TenKy asc
	IF(@IDKyDauTien = @KyCanDongID)
	BEGIN
		SET @LaKyDauTien = 1
	END
	
	Declare @ThangKyCanDong int;
	Declare @NamKyCanDong int;
	Declare @DaDong_KyCanDong bit;
	Declare @TenKyCanDong int;

	Declare @KyIDDaDongGanNhat int;
	Declare @ThangKyDaDongGanNhat int;
	Declare @NamKyDaDongGanNhat int;
	Declare @TenKyDaDongGanNhat int;

	DECLARE @UserNameCanDongKy nvarchar(100)

	Declare @DauKy float;
	Declare @PhatSinhThuDR float;
	Declare @PhatSinhChiCR float;

	Declare @NgayDauKy datetime;
	Declare @NgayCuoiKy datetime;
	Declare @NgayDauKyLienKeTruoc datetime;


	--Lấy thông tin kỳ hiện tại
	SELECT  @TenKyCanDong = TenKy, @DaDong_KyCanDong = DaDong, @NgayDauKy = NgayDauKy, @NgayCuoiKy = NgayCuoiKy
	--,@ThangKyCanDong = Thang, @NamKyCanDong = Nam
	FROM tbKy
	WHERE KyID = @KyCanDongID

	--SET @NgayDauKy = DATETIMEFROMPARTS (@NamKyCanDong, @ThangKyCanDong, 1, 0, 0, 0, 0)
	--SET @NgayCuoiKy = DATEADD(month, 1, @NgayDauKy)
	SET @NgayDauKyLienKeTruoc = DATEADD(month, -1, @NgayDauKy)

	--Lấy kỳ đã đóng gần nhất
	SELECT @KyIDDaDongGanNhat = KyID, @ThangKyDaDongGanNhat = Thang, @NamKyDaDongGanNhat = Nam, @TenKyDaDongGanNhat = TenKy 
	FROM tbKy
	WHERE Nam = YEAR(@NgayDauKyLienKeTruoc) and Thang = MONTH(@NgayDauKyLienKeTruoc)

	-- Nếu kỳ trước chưa đóng thì không làm nữa
	IF(EXISTS (SELECT * FROM tbKy where DaDong = 0 and TenKy < @TenKyCanDong))
	BEGIN
		SELECT 1 -- Có kỳ trước chưa đóng
	END
	-- kiểm tra đã có kỳ liền kề trước chưa
	ELSE IF ((@KyIDDaDongGanNhat IS NULL) AND (@LaKyDauTien = 0))
	BEGIN
		SELECT 2 -- Không có kỳ liền kề trước
	END
	-- Nếu bảng công nợ có dữ liệu chưa duyệt thì không cho đóng kỳ
	ELSE IF (EXISTS (select * from [CONGNO] where DaXoa = 0 and Status <> 1 and NgayGhiNo < @NgayCuoiKy)) --and [NgayGhiNo] > @NgayDauKy
	BEGIN
		SELECT 3 -- Có dữ liệu công nợ chưa duyệt
	END
	ELSE
	BEGIN
		--Lập danh sách user với các thông tin:
			--> Tính giá trị DauKy = CuoiKy của kỳ liền kề trước
				--> nếu kỳ gần nhất là kỳ liền kề thì không cần tính, lấy số cuối kỳ là xong
				--> nếu kỳ gần nhất là khoảng cách nhiều kỳ hoặc không có kỳ đã đóng gần nhất thì -->Tính phát sinh từ NgayDongKyGanNhat đến NgayDauKyCanDong
			--> Tính giá trị phát sinh thu DR, phát sinh chi CR
		--> Lưu vào bảng tbChotKy

		--Xóa dữ liệu cũ
		delete tbChotKy where KyID = @KyCanDongID

		--Thêm dữ liệu mới
		INSERT INTO tbChotKy (KyID, UserName, DauKy, PhatSinhThuDR, PhatSinhChiCR, NguoiTao, NgayTao)
		select 
			@KyCanDongID as KyID
			, U.[UserName]
			, CASE WHEN @LaKyDauTien = 1 THEN ISNULL((select sum([CanDoi]) from [CONGNO] where DaXoa = 0 and Status = 1 and [NgayGhiNo] < @NgayDauKy and UserName = U.UserName), 0)
										ELSE ISNULL((select top 1 CuoiKy from tbChotKy where UserName = U.UserName and KyID = @KyIDDaDongGanNhat), 0)
			END as DauKy
			,SUM(ISNULL([DR], 0)) as PhatSinhThuDR
			,SUM(ISNULL([CR], 0)) as PhatSinhChiCR
			,@NguoiTao as NguoiTao
			,getdate() as NgayTao
		FROM AspNetUsers U
		LEFT JOIN [CONGNO] CN ON U.UserName = CN.UserName and DaXoa = 0 and [Status] = 1 and [NgayGhiNo] between @NgayDauKy and @NgayCuoiKy
		group by U.[UserName]

		UPDATE tbKy SET DaDong = 1 WHERE KyID = @KyCanDongID

		SELECT 0;
	END
END;

CREATE PROCEDURE [dbo].[SP_DongKyMoTam]
	@ChotKyID int
	, @NguoiTao nvarchar(50)
AS
BEGIN
	Declare @UserNameCanDong nvarchar(250);
	Declare @NgayCuoiKyCanDong datetime;

    DECLARE @CS_KyID int;
    DECLARE @CS_NgayDauKy datetime;
    DECLARE @CS_NgayCuoiKy datetime;

	--Lấy thông tin User và ngày cuối kỳ cần đóng
	SELECT @NgayCuoiKyCanDong = ky.NgayCuoiKy, @UserNameCanDong = CK.UserName
	FROM tbChotKy CK
	INNER JOIN tbKy Ky ON CK.KyID = Ky.KyID
	WHERE ChotKyID = @ChotKyID

	DECLARE RS SCROLL CURSOR FOR 
		SELECT Ky.KyID, NgayDauKy, NgayCuoiKy
		FROM tbKy Ky
		INNER JOIN tbChotKy CK ON CK.KyID = Ky.KyID AND CK.UserName = @UserNameCanDong
		WHERE ky.NgayCuoiKy >= @NgayCuoiKyCanDong
		ORDER BY Ky.TenKy

	OPEN RS

	FETCH NEXT FROM RS INTO @CS_KyID, @CS_NgayDauKy, @CS_NgayCuoiKy
	WHILE(@@FETCH_STATUS <> -1)
	BEGIN
		UPDATE tbChotKy
		SET TamMoKy = 0
		, DauKy = ISNULL((select top 1 CuoiKy from tbChotKy where UserName = @UserNameCanDong and KyID = (SELECT top 1 KyID FROM tbKy WHERE Nam = YEAR(DATEADD(month, -1, @CS_NgayDauKy)) and Thang = MONTH(DATEADD(month, -1, @CS_NgayDauKy)))), 0)
		, PhatSinhThuDR = (Select ISNULL(SUM(ISNULL([DR], 0)),0) from [CONGNO] where UserName = @UserNameCanDong and HienTai = 1 and DaXoa = 0 and [Status] = 1 and [NgayGhiNo] between @CS_NgayDauKy and @CS_NgayCuoiKy)
		, PhatSinhChiCR = (Select ISNULL(SUM(ISNULL([CR], 0)),0) from [CONGNO] where UserName = @UserNameCanDong and HienTai = 1 and DaXoa = 0 and [Status] = 1 and [NgayGhiNo] between @CS_NgayDauKy and @CS_NgayCuoiKy)
		, NgayCapNhatCuoi = GETDATE()
		, NguoiCapNhatCuoi = @NguoiTao
		WHERE KyID = @CS_KyID and UserName = @UserNameCanDong

		FETCH NEXT FROM RS INTO @CS_KyID, @CS_NgayDauKy, @CS_NgayCuoiKy
	END
	CLOSE RS
	DEALLOCATE RS

	SELECT 0;
END;

CREATE PROCEDURE [dbo].[SP_KiemTra_DuocCapNhatCongNo]
	@NgayGhiNo datetime
	, @UserName nvarchar(250)
AS
BEGIN
	--0: Được cập nhật
	--1: Không được cập nhật/thêm mới (do chưa khởi tạo kỳ)
	--2: Không được cập nhật/thêm mới (do kỳ đang đóng)

	DECLARE @DaDong bit
	DECLARE @DuocCapNhat int
	SET @DuocCapNhat = 1 

	DECLARE @KyID int

	SELECT @KyID = KyID, @DaDong = DaDong
	FROM tbKy 
	WHERE Nam = YEAR(@NgayGhiNo)
	and Thang = MONTH(@NgayGhiNo) 
	--and DaDong = 1

	IF (@KyID Is NULL)
	BEGIN
		SET @DuocCapNhat = 1 -- Không được cập nhật/thêm mới (do chưa khởi tạo kỳ)
	END
	ELSE -- Đã khởi tạo kỳ
	BEGIN
		IF(@DaDong = 0) --Kỳ đang mở
		BEGIN
			SET @DuocCapNhat = 0 -- Được cập nhật
		END
		ELSE--Kỳ đã đóng
		BEGIN
			IF((@UserName <> '') AND (EXISTS (SELECT * FROM tbChotKy where KyID = @KyID and UserName = @UserName and TamMoKy = 1)))
			BEGIN
				SET @DuocCapNhat = 0 -- Được cập nhật do tạm mở cho user
			END
			ELSE
			BEGIN
				SET @DuocCapNhat = 2 -- Không được cập nhật/thêm mới (do kỳ đang đóng)
			END
		END
	END

	RETURN @DuocCapNhat
END;

CREATE PROCEDURE [dbo].[SP_KiemTra_DuocCapNhatDonHang]
	@ID int
	, @UserName nvarchar(250)
AS
BEGIN
	DECLARE	@DuocCapNhat int
	DECLARE @usernameOld nvarchar(50)
	DECLARE	@NgayGhiNoOld datetime
	DECLARE	@DuocCapNhatOld int
	SET @DuocCapNhatOld = 0
	
	SELECT @usernameOld = username FROM DON_HANG WHERE  ID = @ID

	
	IF(@UserName <> @usernameOld)--Nếu có đổi user, xem TẤT CẢ công nợ của user cũ (mua hàng, ship, cân,....) có bị khóa không
	BEGIN
		DECLARE @CS_NgayGhiNo datetime;
		DECLARE	@DuocCapNhatOld1 int
		DECLARE	@DuocCapNhatOld2 int
		SET @DuocCapNhatOld1 = 0
		SET @DuocCapNhatOld2 = 0

		DECLARE RS SCROLL CURSOR FOR 
		SELECT NgayGhiNo FROM CONGNO WHERE DonHang_ID = @ID AND style = 1

		OPEN RS

		FETCH NEXT FROM RS INTO @CS_NgayGhiNo
		WHILE(@@FETCH_STATUS <> -1)
		BEGIN
			EXEC @DuocCapNhatOld1 = [SP_KiemTra_DuocCapNhatCongNo] @CS_NgayGhiNo, @usernameOld
			EXEC @DuocCapNhatOld2 = [SP_KiemTra_DuocCapNhatCongNo] @CS_NgayGhiNo, @UserName
			--IF(@DuocCapNhatOld1 <> 0)
			IF ((@DuocCapNhatOld1 <> 0) OR (@DuocCapNhatOld2 <> 0))
			BEGIN
				SET @DuocCapNhatOld = 2
			END

			FETCH NEXT FROM RS INTO @CS_NgayGhiNo
		END
		CLOSE RS
		DEALLOCATE RS
	END
	ELSE IF(EXISTS(SELECT * FROM CONGNO WHERE HienTai = 1 AND DonHang_ID = @ID AND style = 1)) --Nếu không đổi user thì chỉ cần kiểm tra công nợ mua hàng của user đó
	BEGIN
		SELECT @usernameOld = username, @NgayGhiNoOld = NgayGhiNo FROM CONGNO WHERE HienTai = 1 AND DonHang_ID = @ID AND style = 1
		EXEC @DuocCapNhatOld1 = [SP_KiemTra_DuocCapNhatCongNo] @NgayGhiNoOld, @usernameOld
		EXEC @DuocCapNhatOld2 = [SP_KiemTra_DuocCapNhatCongNo] @NgayGhiNoOld, @UserName
		--IF(@DuocCapNhatOld1 <> 0)
		IF ((@DuocCapNhatOld1 <> 0) OR (@DuocCapNhatOld2 <> 0))
		BEGIN
			SET @DuocCapNhatOld = 2
		END
	END
	

	DECLARE	@NgayGhiNo datetime
	DECLARE	@DuocCapNhat2 int
	SET @NgayGhiNo = GETDATE()

	--kiểm tra xem user mới và ngày ghi nợ mới có bị khóa không
	EXEC @DuocCapNhat2 = [SP_KiemTra_DuocCapNhatCongNo] @NgayGhiNo, @username

	IF ((@DuocCapNhat2 = 0) AND (@DuocCapNhatOld = 0))
	BEGIN
		SET @DuocCapNhat = 0;
	END
	ELSE
	BEGIN
		SET @DuocCapNhat = 1;
	END

	RETURN @DuocCapNhat;
END;

CREATE PROCEDURE [dbo].[SP_KiemTra_HanMucKhachHang]
	@username nvarchar(100)
AS
BEGIN
	DECLARE @TongTienDatHang float;
	DECLARE @TongTienKhachCo float;
	DECLARE @TongTienKhachNo float;
	DECLARE @TongTienOrderHang float;
	IF(NOT EXISTS (SELECT * FROM DON_HANG WHERE DaXoa = 0 AND username = @username AND trangthaiOrder = 'Completed'))
	BEGIN
		
		SELECT @TongTienDatHang = SUM(CASE WHEN tongtienVND is not null THEN tongtienVND ELSE soluong*dongiaweb*tygia END) FROM DON_HANG WHERE DaXoa = 0 AND username = @username AND trangthaiOrder = 'Received';
		SELECT @TongTienKhachCo = SUM(CanDoi) * (-1) FROM CONGNO WHERE DaXoa = 0 AND UserName = @username;

		IF(@TongTienKhachCo > @TongTienDatHang * 0.8)
		BEGIN
			exec [SP_CapNhat_HanMucKhachHangTheoUser] @username, 0
		END
		ELSE
		BEGIN
			exec [SP_CapNhat_HanMucKhachHangTheoUser] @username, 1			
		END
	END
	ELSE
	BEGIN
		SELECT @TongTienOrderHang = SUM(tongtienVND) FROM DON_HANG WHERE DaXoa = 0 AND username = @username AND trangthaiOrder = 'Ordered' AND ngayveVN is null;
		SELECT @TongTienKhachNo = SUM(CanDoi) FROM CONGNO WHERE DaXoa = 0 AND UserName = @username;

		IF(@TongTienKhachNo < @TongTienOrderHang * 0.3)
		BEGIN
			exec [SP_CapNhat_HanMucKhachHangTheoUser] @username, 0
		END
		ELSE
		BEGIN
			exec [SP_CapNhat_HanMucKhachHangTheoUser] @username, 1			
		END
	END
END;

						END as MoTaTinhTrang

	WHERE TT.TrackingID = @TrackingID
	AND TT.DaXoa = 0
	ORDER BY NgayChuyenTinhTrang desc

AS
BEGIN

			, DH2.MaDatHang, DH2.TienHangUSD, DH2.TienHangVND 
	FROM DotHang T
	INNER JOIN (
				SELECT username, TenDotHang, SUM(tongtienUSD) as TienHangUSD, SUM(tongtienVND) as TienHangVND
				, STUFF((
						SELECT ',' + CAST(ID as nvarchar(10))
						FROM DON_HANG DH3

				) DH2 ON T.UserName = DH2.username AND T.TenDotHang = DH2.TenDotHang
	WHERE ShipperID is null AND YeuCauGuiHang = 1
	ORDER BY T.TenDotHang desc

			 , GH.ShipperID
			 , DH.TenDotHang
			 , GH.SoVanDon
			 , GH.PhiShipTrongNuoc
		From DON_HANG DH
		INNER JOIN DotHang GH ON DH.username = GH.UserName and DH.TenDotHang = GH.TenDotHang and GH.ShipperID is not null
		where DaXoa = 0 
		and trangthaiOrder = 'Shipped' 
		and DH.username = CASE WHEN @UserName = '' THEN DH.username ELSE @UserName END
		group by DH.username, GH.ShipperID, DH.TenDotHang, GH.SoVanDon, GH.PhiShipTrongNuoc
		order by DH.TenDotHang, DH.username desc

	select MAX(#tbTemp.STT) from #tbTemp ;

	SELECT T.*, S.ShipperName 
	FROM #tbTemp T
	INNER JOIN SHIPPER_MAN S ON T.ShipperID = S.ID
	WHERE STT between (@PageNum-1)*@PageSize+1 and (@PageNum-1)*@PageSize + @PageSize
	ORDER BY STT
		
	DROP TABLE #tbTemp;
END;

CREATE PROCEDURE [dbo].[SP_Lay_DotHangShip]
	@username nvarchar(100)
	,@YeuCauGuiHang int
	--,@PageSize int
	--,@PageNum int
AS
BEGIN
	SELECT T.*
			, DH2.MaDatHang, DH2.TienHangUSD, DH2.TienHangVND 
	FROM DotHang T
	INNER JOIN (
				SELECT TenDotHang, SUM(tongtienUSD) as TienHangUSD, SUM(tongtienVND) as TienHangVND
				, STUFF((
						SELECT ',' + CAST(ID as nvarchar(10))
						FROM DON_HANG 
						WHERE (TenDotHang = DH.TenDotHang AND username = @username) 
						FOR XML PATH(''),TYPE).value('(./text())[1]','VARCHAR(MAX)')
						,1,1,'') AS MaDatHang
				FROM DON_HANG DH
				WHERE trangthaiOrder = 'Ordered' AND ngayveVN is not null AND username = @username
				GROUP BY TenDotHang
				) DH2 ON T.TenDotHang = DH2.TenDotHang
	WHERE 1 = 1 
	AND T.username = @UserName
	AND T.YeuCauGuiHang = @YeuCauGuiHang
	ORDER BY T.TenDotHang desc
END;

	,@PageNum int

BEGIN

    DECLARE @start int;
    DECLARE @run int;
    SET @start=(@PageNum - 1)*@PageSize+1;
    SET @run=0;

CREATE PROCEDURE [dbo].[SP_Lay_Tracking]
	@UserName nvarchar(250)
	,@TinhTrang nvarchar(100)
	,@NoiDungTim nvarchar(250)
	,@TimTheo int
	,@TrackingNumber nvarchar(500)
	,@TenLoHang nvarchar(50)
	,@PageSize int
	,@PageNum int
	,@QuocGiaID int
	,@DaXoa bit
	,@TuNgay nvarchar(15)
	,@DenNgay nvarchar(15)
AS
BEGIN
	CREATE TABLE #MyPage(TrackingID int);
    DECLARE @selectitem int;
    DECLARE @start int;
    DECLARE @run int;
    SET @start=(@PageNum - 1)*@PageSize+1;
    SET @run=0;
   
	DECLARE @Sql nvarchar(MAX);
    SET @Sql = 'DECLARE RS SCROLL CURSOR FOR ';
		
	SET @Sql = @Sql + ' SELECT TrackingID FROM tbTracking TK LEFT JOIN tbQuocGia QG ON TK.QuocGiaID = QG.QuocGiaID WHERE DaXoa = ' + CAST(@DaXoa AS CHAR(1))
	
	IF(@UserName <> '')
	BEGIN
		SET @Sql = @Sql + ' AND UserName = ''' + @UserName + ''''
	END
	
	IF(@TinhTrang <> '')
	BEGIN
		SET @Sql = @Sql + ' AND TinhTrang in (' + @TinhTrang + ')'
	END
	
	IF(@NoiDungTim <> '')
	BEGIN
		IF(@TimTheo = -1)
		BEGIN
			SET @Sql = @Sql + ' AND ((TrackingID like N''%' + @NoiDungTim + '%'') OR (OrderNumber like N''%' + @NoiDungTim + '%'') OR (TenLoHang like N''%' + @NoiDungTim + '%'') OR (TrackingNumber like N''%' + @NoiDungTim + '%'') OR (GhiChu like N''%' + @NoiDungTim + '%'') OR (Kien like N''%' + @NoiDungTim + '%'') OR (Mawb like N''%' + @NoiDungTim + '%'') OR (Hawb like N''%' + @NoiDungTim + '%'') OR (TenQuocGia like N''%' + @NoiDungTim + '%''))'
		END
		ELSE IF(@TimTheo = 0)
		BEGIN
			SET @Sql = @Sql + ' AND (OrderNumber like N''%' + @NoiDungTim + '%'')'
		END
	END
	
	IF(@TrackingNumber <> '')
	BEGIN
		SET @Sql = @Sql + ' AND TrackingNumber like ''%' + @TrackingNumber + '%'''
	END

	IF(@TenLoHang <> '')
	BEGIN
		SET @Sql = @Sql + ' AND TenLoHang = ''' + @TenLoHang + ''''
	END

	IF(@QuocGiaID <> -1)
	BEGIN
		SET @Sql = @Sql + ' AND TK.QuocGiaID = ' + CAST(@QuocGiaID as nvarchar(10))
	END

	IF(@TuNgay <> '')
	BEGIN
		SET @Sql = @Sql + ' AND NgayDatHang >= ''' + @TuNgay + ''''
	END

	IF(@DenNgay <> '')
	BEGIN
		SET @Sql = @Sql + ' AND NgayDatHang <= ''' + @DenNgay+ ''''
	END

	SET @Sql = @Sql + ' ORDER BY NgayDatHang desc, TrackingID desc'
	
	--select @Sql
	
	EXEC sp_executesql @Sql;
	
	OPEN RS
	--return total record
	SELECT @@CURSOR_ROWS AS TOTALROW;
	---------
	FETCH ABSOLUTE @start FROM RS INTO @selectitem
	WHILE(@@FETCH_STATUS <> -1 and @run < @PageSize)
	BEGIN
		INSERT INTO #MyPage(TrackingID) VALUES(@selectitem)
		SET @run=@run+1;
		FETCH NEXT FROM RS INTO @selectitem
	END
	CLOSE RS
	DEALLOCATE RS
	
	SELECT TK.*, VC.TenNhaVanChuyen, QG.TenQuocGia, (select count(*) from tbChiTietTracking where TrackingID = #MyPage.TrackingID) as SoLuongChiTiet, U.LinkTaiKhoanMang
	FROM tbTracking TK
	INNER JOIN #MyPage ON TK.TrackingID = #MyPage.TrackingID
	LEFT JOIN tbNhaVanChuyen VC ON TK.NhaVanChuyenID = VC.NhaVanChuyenID
	LEFT JOIN tbQuocGia QG ON TK.QuocGiaID = QG.QuocGiaID
	LEFT JOIN AspNetUsers U ON TK.username = U.UserName

END;

CREATE PROCEDURE [dbo].[SP_ShareOrders]
	@id nvarchar(4000)
	,@ordernumber nvarchar(50)
	,@saleoff float
	,@phuthu float
	,@shipUSA float
	,@tax float
	,@cong float
	,@TotalCharged float
	,@TotalItem int
	,@HeThongTuTinhCong bit
	,@tygia float
	,@NguoiTao nvarchar(250)
AS
BEGIN TRY
	SET NOCOUNT ON;
	BEGIN TRANSACTION
	DECLARE	@NgayGhiNo datetime
	SET @NgayGhiNo = GETDATE();

	DECLARE	@DuocCapNhat int
	EXEC @DuocCapNhat = [SP_KiemTra_DuocCapNhatCongNo] @NgayGhiNo, ''

	IF (@DuocCapNhat = 0)
	BEGIN
		DECLARE @Thue float
		SELECT @Thue = CAST(GiaTriCauHinh as float) FROM CauHinh WHERE TenCauHinh = 'Thue'

		DECLARE @id_item nvarchar(50)
		DECLARE @soluong int
		DECLARE @dongiaweb float
		--DECLARE @tygia float
		DECLARE @loaitien nvarchar(50)
		DECLARE @tien float
		DECLARE @username nvarchar(50)
		DECLARE @tiencong float

		DECLARE id_cursor CURSOR FOR  SELECT s FROM dbo.SplitString(@id, ',')
		OPEN id_cursor
		FETCH NEXT FROM id_cursor INTO @id_item
		WHILE @@FETCH_STATUS = 0
		BEGIN
			SELECT @soluong= soluong, @dongiaweb = dongiaweb, @username = username, @loaitien = loaitien
			FROM DON_HANG
			WHERE ID = CAST(@id_item as int)

			SET @tien = ((100 - @saleoff) / 100) * @dongiaweb * @soluong;

			SET @tiencong = @tien * @cong / 100;


			DECLARE @tiencongVND float
			DECLARE @tiencong1mon float
			DECLARE @tinhtheophantram float
			DECLARE @khachbuon bit
			SET @tiencongVND = 0;

			SELECT @khachbuon = KhachBuon FROM AspNetUsers where UserName = @username
			SELECT @tiencong1mon=TienCong1Mon, @tinhtheophantram = TinhTheoPhanTram FROM GiaTienCong WHERE ((@dongiaweb  > TuGia) and (@dongiaweb  <= DenGia)) and LoaiTien = @loaitien and KhachBuon = @khachbuon
        
			IF(@HeThongTuTinhCong=1)
			BEGIN
				IF (@tinhtheophantram=1)
				BEGIN
					SET @cong = @tiencong1mon;
					SET @tiencong = @tien * @cong / 100;
					SET @tiencongVND = @tiencong * @tygia;
				END
				ELSE
				BEGIN
					SET @tiencong = 0;
					SET @tiencongVND = @tiencong1mon * @soluong;
				END
			END
			ELSE
			BEGIN
					SET @tiencong = @tien * @cong / 100;
					SET @tiencongVND = @tiencong * @tygia;
			END
		
			SET @tien = @tien + @tien * 0.01 * @tax;

			SET @tien = @tien + @shipUSA * @soluong;

			SET @tien = @tien + @phuthu * @soluong;

			UPDATE DON_HANG
			SET saleoff = @saleoff
			, tygia = @tygia
			, phuthu = @phuthu
			, shipUSA = @shipUSA
			, tax = @tax
			, cong = @cong
			, giasauoffUSD =  @tien
			, giasauoffVND = @tien * tygia
			, tiencongUSD = @tiencong
			--, tiencongVND = @tiencong * tygia
			, tiencongVND = @tiencongVND
			, tongtienUSD = @tien + @tiencong
			, tongtienVND = CAST((CEILING(((@tien * @tygia) + @tiencongVND) * 1) / 1) as int)
			, ordernumber = @ordernumber
			, trangthaiOrder = 'Ordered'
			WHERE ID = CAST(@id_item as int)
		
			IF(EXISTS(SELECT * FROM CONGNO WHERE HienTai = 1 AND DonHang_ID = @id_item AND style = 1))
			BEGIN
				-- update CONGNO
				UPDATE CONGNO
				SET DR = CAST((CEILING(((@tien * @tygia) + @tiencongVND) * 1) / 1) as int)
				, UserName = @username
				, DaXoa = 0
				, NguoiCapNhatCuoi = @NguoiTao
				, NgayCapNhatCuoi = GETDATE()
				WHERE HienTai = 1 AND DonHang_ID = @id_item AND style = 1
			END
			ELSE
			BEGIN
			-- insert CONGNO
				INSERT INTO CONGNO(DonHang_ID, NoiDung, NgayGhiNo, DR, CR, UserName, [Status], style, HienTai, NguoiTao)
				VALUES (@id_item, N'Phí mua ' + @id_item, GETDATE(), CAST((CEILING(((@tien * @tygia) + @tiencongVND) * 1) / 1) as int), 0, @username, 1, 1, 1, @NguoiTao)
			END
	
			FETCH NEXT FROM id_cursor INTO @id_item
		END 
		CLOSE id_cursor;
		DEALLOCATE id_cursor;

		IF(EXISTS(SELECT * FROM LIST_ORDER WHERE OrderNumber = @ordernumber))
		BEGIN
			-- update LIST_ORDER
			UPDATE LIST_ORDER
			SET TongSoLink = @TotalItem
			, TongTienOrder = @TotalCharged
			, TongTienOrderVND = CAST((CEILING(((@TotalCharged * @tygia)) * 1) / 1) as int)
			WHERE OrderNumber = @ordernumber
		END
		ELSE
		BEGIN
			-- insert LIST_ORDER
			INSERT INTO LIST_ORDER (OrderNumber, TongSoLink, TongTienOrder, TongTienOrderVND)
			VALUES (@ordernumber, @TotalItem, @TotalCharged, CAST((CEILING(((@TotalCharged * @tygia)) * 1) / 1) as int))
		END
	END

COMMIT TRANSACTION
END TRY
BEGIN CATCH
	-- Whoops, there was an error
	 IF @@TRANCOUNT > 0
	 ROLLBACK TRANSACTION
	-- Raise an error with the details of the exception
	 DECLARE @ErrMsg nvarchar(4000), @ErrSeverity int

	 SELECT @ErrMsg = ERROR_MESSAGE(),
	 @ErrSeverity = ERROR_SEVERITY()

	 RAISERROR(@ErrMsg, @ErrSeverity, 1)
END CATCH;

CREATE PROCEDURE [dbo].[SP_Sua_Ky]
	@KyID int
	, @Thang int
	, @Nam int
AS
BEGIN
	Declare @NgayDauKy datetime;
	Declare @NgayCuoiKy datetime;

	--Lấy thông tin kỳ hiện tại
	SELECT @NgayDauKy = NgayDauKy, @NgayCuoiKy = NgayCuoiKy
	FROM tbKy
	WHERE KyID = @KyID

	IF(EXISTS (SELECT * FROM tbKy where Nam = @Nam and Thang = @Thang))
	BEGIN
		SELECT 1 -- Đã có kỳ rồi
	END
	ELSE IF(EXISTS (SELECT *  FROM CONGNO WHERE NgayGhiNo between @NgayDauKy and @NgayCuoiKy))
	BEGIN
		SELECT 2 -- Kỳ đã phát sinh dữ liệu
	END
	ELSE
	BEGIN
		UPDATE tbKy
		SET Thang = @Thang
		, Nam = @Nam
		WHERE KyID = @KyID

		SELECT 0;
	END
END;

CREATE PROCEDURE [dbo].[SP_TamMoKy]
	@ChotKyID int
AS
BEGIN
	Declare @DaDong_KyCanMo bit;
	Declare @TenKyCanMo int;
	Declare @NgayDauKy datetime;
	Declare @NgayCuoiKy datetime;

	--Lấy thông tin kỳ cần mở
	SELECT  @TenKyCanMo = ky.TenKy, @DaDong_KyCanMo = ky.DaDong, @NgayDauKy = ky.NgayDauKy, @NgayCuoiKy = ky.NgayCuoiKy
	FROM tbChotKy CK
	INNER JOIN tbKy Ky ON CK.KyID = Ky.KyID
	WHERE ChotKyID = @ChotKyID

	IF(@NgayCuoiKy < DATEADD(month,-2, GETDATE()))
	BEGIN
		SELECT 1 -- Không được tạm mở kỳ quá xa
	END
	ELSE
	BEGIN
		UPDATE tbChotKy SET TamMoKy = 1 WHERE ChotKyID = @ChotKyID
		SELECT 0;
	END
END;

CREATE PROCEDURE [dbo].[SP_TimKiemSystemLogs]
    @NguoiTao nvarchar(50)
    ,@Nguon nvarchar(50)
    ,@HanhDong nvarchar(250)
    ,@DoiTuong nvarchar(50)
    ,@TuNgay nvarchar(50)
    ,@DenNgay nvarchar(50)
    ,@NoiDung nvarchar(500)
    ,@PageSize int
	,@PageNum int
AS
BEGIN
	CREATE TABLE #MyPage(SystemLogsID uniqueidentifier);
    DECLARE @selectitem uniqueidentifier;
    DECLARE @start int;
    DECLARE @run int;
    SET @start=(@PageNum - 1)*@PageSize+1;
    SET @run=0;
    
    
    DECLARE @Sql nvarchar(MAX);
	SET @Sql = 'DECLARE RS SCROLL CURSOR FOR SELECT SystemLogsID FROM tbSystemLogs WHERE 1 = 1'
	
	IF(@NguoiTao <> '')
	BEGIN
		SET @Sql = @Sql + ' AND NguoiTao like ''%' + @NguoiTao + '%'''
	END
	
	IF(@Nguon <> '')
	BEGIN
		SET @Sql = @Sql + ' AND Nguon like N''%' + @Nguon + '%'''
	END
	
	IF(@HanhDong <> '')
	BEGIN
		SET @Sql = @Sql + ' AND HanhDong like N''%' + @HanhDong + '%'''
	END
	
	IF(@DoiTuong <> '')
	BEGIN
		SET @Sql = @Sql + ' AND DoiTuong like N''%' + @DoiTuong + '%'''
	END
	
	IF(@TuNgay <> '')
	BEGIN
		SET @Sql = @Sql + ' AND NgayTao > ''' + @TuNgay + ''''
	END
	
	IF(@DenNgay <> '')
	BEGIN
		SET @Sql = @Sql + ' AND NgayTao < ''' + @DenNgay + ''''
	END
	
	IF(@NoiDung <> '')
	BEGIN
		SET @Sql = @Sql + ' AND NoiDung like N''%' + @NoiDung + '%'''
	END

	SET @Sql = @Sql + ' ORDER BY NgayTao desc'
		
	EXEC sp_executesql @Sql;
	
	OPEN RS
    --return total record
    SELECT @@CURSOR_ROWS AS TOTALROW;
    ---------
    FETCH ABSOLUTE @start FROM RS INTO @selectitem
    WHILE(@@FETCH_STATUS <> -1 and @run < @PageSize)
    BEGIN
        INSERT INTO #MyPage(SystemLogsID) VALUES(@selectitem)
        SET @run=@run+1;
        FETCH NEXT FROM RS INTO @selectitem
    END
    CLOSE RS
    DEALLOCATE RS
    
    SELECT SL.*
	FROM tbSystemLogs SL
	INNER JOIN #MyPage ON SL.SystemLogsID = #MyPage.SystemLogsID
	ORDER BY  NgayTao desc
END;

CREATE PROCEDURE [dbo].[SP_Xoa_MassDelete]
	@id nvarchar(4000)
	,@NguoiTao nvarchar(250)
AS
BEGIN TRY
	SET NOCOUNT ON;
	BEGIN TRANSACTION

	DECLARE @usernameOld nvarchar(50)
	DECLARE	@NgayGhiNoOld datetime
	DECLARE	@DuocCapNhatOld int
	SET @DuocCapNhatOld = 0
	DECLARE @CS_DonHangID datetime;
	
	DECLARE RS_DonHang SCROLL CURSOR FOR  SELECT CAST(s as int) FROM dbo.SplitString(@id, ',')
	OPEN RS_DonHang

	FETCH NEXT FROM RS_DonHang INTO @CS_DonHangID
	WHILE(@@FETCH_STATUS <> -1)
	BEGIN

		SELECT @usernameOld = username FROM DON_HANG WHERE  ID = @CS_DonHangID

		DECLARE @CS_NgayGhiNo datetime;
		DECLARE	@DuocCapNhatOld1 int
		SET @DuocCapNhatOld1 = 0

		DECLARE RS SCROLL CURSOR FOR 
		SELECT NgayGhiNo FROM CONGNO WHERE DonHang_ID = @ID

		OPEN RS

		FETCH NEXT FROM RS INTO @CS_NgayGhiNo
		WHILE(@@FETCH_STATUS <> -1)
		BEGIN
			EXEC @DuocCapNhatOld1 = [SP_KiemTra_DuocCapNhatCongNo] @CS_NgayGhiNo, @usernameOld
			IF(@DuocCapNhatOld1 <> 0)
			BEGIN
				SET @DuocCapNhatOld = 2
			END

			FETCH NEXT FROM RS INTO @CS_NgayGhiNo
		END
		CLOSE RS
		DEALLOCATE RS

		FETCH NEXT FROM RS_DonHang INTO @CS_DonHangID
	END
	CLOSE RS_DonHang
	DEALLOCATE RS_DonHang

	IF (@DuocCapNhatOld = 0)
	BEGIN
		UPDATE DON_HANG SET DaXoa = 1 WHERE ID IN (SELECT CAST(s as int) FROM dbo.SplitString(@id, ','))

		UPDATE CONGNO
		SET DaXoa = 1
		, HienTai = 0
		, NguoiCapNhatCuoi = @NguoiTao
		, NgayCapNhatCuoi = GETDATE()
		WHERE DonHang_ID IN (SELECT CAST(s as int) FROM dbo.SplitString(@id, ','))
	END

	COMMIT TRANSACTION
END TRY
BEGIN CATCH
	-- Whoops, there was an error
		IF @@TRANCOUNT > 0
		ROLLBACK TRANSACTION
	-- Raise an error with the details of the exception
		DECLARE @ErrMsg nvarchar(4000), @ErrSeverity int

		SELECT @ErrMsg = ERROR_MESSAGE(),
		@ErrSeverity = ERROR_SEVERITY()

		RAISERROR(@ErrMsg, @ErrSeverity, 1)
END CATCH;

CREATE function [dbo].[SplitString] 
(
    @str nvarchar(4000), 
    @separator char(1)
)
RETURNS  
@ReturnVal TABLE 
(
    zeroBasedOccurance int
	,s nvarchar(4000)
)
AS
BEGIN
	
	with tokens(p, a, b) AS (
		select 
			1, 
			1, 
			charindex(@separator, @str)
		union all
		select
			p + 1, 
			b + 1, 
			charindex(@separator, @str, b + 1)
		from tokens
		where b > 0
	)
	insert into @ReturnVal(zeroBasedOccurance,s)
	select
		p-1 zeroBasedOccurance,
		substring(
			@str, 
			a, 
			case when b > 0 then b-a ELSE 4000 end) 
		AS s
	from tokens
	OPTION (MAXRECURSION 0)

	RETURN
END;

CREATE function [dbo].[SplitString1] 
(
    @str nvarchar(4000), 
    @separator char(1)
)
returns table
AS
return (
    with tokens(p, a, b) AS (
        select 
            1, 
            1, 
            charindex(@separator, @str)
        union all
        select
            p + 1, 
            b + 1, 
            charindex(@separator, @str, b + 1)
        from tokens
        where b > 0
    )
    select
        p-1 zeroBasedOccurance,
        substring(
            @str, 
            a, 
            case when b > 0 then b-a ELSE 4000 end) 
        AS s
    from tokens
  );
